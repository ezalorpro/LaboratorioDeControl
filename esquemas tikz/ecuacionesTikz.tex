\begin{tcolorbox}
    Forma general:
    
    \begin{equation}
        \frac{a^n}{a^m} = a^{n-m}
    \end{equation}
        
    Ejemplo:
    
    \begin{align}
        \frac{a^{-2}}{a^4} &= a^{-2-4} \\ 
        &= a^{-6} \\ 
        &= \frac{1}{a^6} 
    \end{align}
\end{tcolorbox}
